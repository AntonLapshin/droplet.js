How it works.

1. Create your first simple component in {your project}/droplets/item/

view.html

<h1 data-bind="text: data.name"></h1>
<p data-bind="text: data.desc"></p>

style.css

.item { }

logic.js

Droplet.define('item', [], function(){
    var self = this;

    self.on('load', function(data){
    });

    self.on('set', function(data){
    });

    self.on('test', function(){
        var data = {
            name: 'Test header',
            desc: 'Test description'
        };
        self.set(data).show();
    });
});

2. Include js files {your project}/index.html and run Droplet

<div id="container"></div>

<script src="lib/jquery.min.js"></script>
<script src="lib/droplet.js"></script>
<script src="lib/droplet-bind.js"></script>

<script>
    Droplet.run('#container');
</script>

3. Open {url to your site}/?item

Reuse your components. Nested components

1. Create your second simple component in {your project}/droplets/gallery/

view.html

<h1 data-bind="text: data.title"></h1>
<div data-droplet="item"></div>

style.css

.gallery {
  .item { }
}

logic.js

Droplet.define('gallery', [], function(){
    var self = this;

    self.on('load', function(data){
    });

    self.on('set', function(data){
        self.children.item.set(data.item).show();
    });

    self.on('test', function(){
        var data = {
            title: 'Test Gallery',
            item: {
                name: 'Test header',
                desc: 'Test description'
            }
        };
        self.set(data).show();
    });
});

3. Open {url to your site}/?gallery
